# 1. Define theme and type of data
theme = "base"
ot_type = "land_use"

# 2. Spatial query, geometery in binary format
rel = con.sql(f"""
    SELECT id,
           class,
           subtype,
           ST_AsWKB(geometry) as geom_binary -- Renamed to avoid confusion
    FROM read_parquet(
    format('s3://overturemaps-us-west-2/release/{{}}/theme={theme}/type={ot_type}/*', getvariable('latest_release')),
    filename=true, hive_partitioning=1)
    WHERE
        bbox.xmin BETWEEN 34.8617 AND 34.9351 AND
        bbox.ymin BETWEEN 32.1311 AND 32.1834;
""").df()
Code 
# Convert the df with geom_binary to gdf

rel['geom_binary'] = rel['geom_binary'].apply(bytes)

hh_landuse = gpd.GeoDataFrame(
    rel.drop(columns=['geom_binary']),
    geometry=gpd.GeoSeries.from_wkb(rel['geom_binary']),
    crs="EPSG:4326"
)

# Spatial query with un-changed geometry - JSON format
# IMPORTANT: This query have the best! filtering

rel2 = con.sql("""
    SELECT
        id,
        class,
        subtype,
        geometry
    FROM read_parquet(
        's3://overturemaps-us-west-2/release/2026-03-18.0/theme=base/type=land_use/*',
        filename=true,
        hive_partitioning=1
    )
    WHERE 
    -- Step 1: Use the bbox for lightning-fast pruning (The "Coarse" filter)
    bbox.xmin <= 34.9351 AND bbox.xmax >= 34.8617 AND
    bbox.ymin <= 32.1834 AND bbox.ymax >= 32.1311
    -- Step 2: Only run the expensive math on the few rows that pass Step 1 (The "Fine" filter)
    AND ST_Intersects(geometry, ST_MakeEnvelope(34.8617, 32.1311, 34.9351, 32.1834));
""")
Code 
# Convert the duckdb.query into gdf with 'arrow'

arrow_table = rel2.arrow()

gdf_landuse = gpd.GeoDataFrame.from_arrow(arrow_table)
